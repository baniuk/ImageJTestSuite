1   package com.github.baniuk.ImageJTestSuite.tools.files;

3   import java.io.IOException;
4   import java.nio.charset.StandardCharsets;
5   import java.nio.file.Files;
6   import java.nio.file.Path;
7   import java.util.List;
8   
9   /**
10   * Modify files or their parts.
11   * 
12   * @author p.baniukiewicz
13   *
14   */
15  public class FileModifiers {
16  
17    /**
18     * Replace line in file.
19     * 
20     * @param name file name and path
21     * @param lineNumber line number from 0
22     * @param data replace string
23     * @throws IOException error
24     */
25    public static void replaceLine(Path name, int lineNumber, String data) throws IOException {
26      List<String> lines = Files.readAllLines(name, StandardCharsets.UTF_8);
27      lines.set(lineNumber, data);
28      Files.write(name, lines, StandardCharsets.UTF_8);
29    }
30  
31    /**
32     * Get line from file.
33     * 
34     * @param name file name and path
35     * @param lineNumber line number from 0
36     * @return Line at position lineNumber
37     * @throws IOException error
38     */
39    public static String getLine(Path name, int lineNumber) throws IOException {
40      List<String> lines = Files.readAllLines(name, StandardCharsets.UTF_8);
41      return lines.get(lineNumber);
42    }
43  
44  }
