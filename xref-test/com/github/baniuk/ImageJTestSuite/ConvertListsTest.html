22    /**
23     * Test method for
24     * {@link com.github.baniuk.ImageJTestSuite.ConvertLists#fromList2Polygon(java.util.List)}.
25     * 
26     * @throws Exception
27     */
28    @Test
29    public void testFromList2Polygon() throws Exception {
30      // empty list
31      List<Point2d> test = new ArrayList<>();
32      FloatPolygon ret = ConvertLists.fromList2Polygon(test);
33      assertThat(ret.npoints, is(0)); // empty polygon
34  
35      // not empty list, check order
36      test.clear();
37      test.add(new Point2d(0, 0));
38      test.add(new Point2d(5, 4));
39      ret = ConvertLists.fromList2Polygon(test);
40      assertThat(ret.npoints, is(2));
41      assertThat(ret.xpoints[1], is(5f));
42      assertThat(ret.ypoints[1], is(4f));
43    }

45    /**
46     * Test method for
47     * {@link com.github.baniuk.ImageJTestSuite.ConvertLists#fromPolygon2List(ij.process.FloatPolygon)}.
48     * 
49     * @throws Exception
50     */
51    @Test
52    public void testFromPolygon2List() throws Exception {
53      FloatPolygon p = new FloatPolygon(); // empty polygon
54      List<Point2d> ret = ConvertLists.fromPolygon2List(p);
55      assertThat(ret.isEmpty(), is(true)); // empty list on return
56  
57      // not empty polygon, check order
58      float[] x = { 0f, 5f };
59      float[] y = { 0f, 4f };
60      p = new FloatPolygon(x, y);
61      ret = ConvertLists.fromPolygon2List(p);
62      assertThat(ret.size(), is(2));
63      assertThat(ret.get(1).x, is(5.0));
64      assertThat(ret.get(1).y, is(4.0));
65    }
